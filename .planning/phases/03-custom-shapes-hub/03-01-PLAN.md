```typescript
import type { TLShape } from 'tldraw'

// Shape type constants
export const SHAPE_TYPES = {
  HUB: 'hub-shape',
  YOUTUBE: 'youtube-node',
  BLOG: 'blog-node',
  PROJECT: 'project-node',
  MILESTONE: 'milestone-node',
} as const

// Module augmentation for tldraw type system
declare module 'tldraw' {
  export interface TLGlobalShapePropsMap {
    [SHAPE_TYPES.HUB]: {
      w: number
      h: number
      name: string
      title: string
      bio: string
      avatar?: string
      social?: {
        github?: string
        twitter?: string
        linkedin?: string
        youtube?: string
      }
    }
    [SHAPE_TYPES.YOUTUBE]: {
      w: number
      h: number
      nodeId: string
      title: string
      url: string
      thumbnail?: string
      date: string
    }
    [SHAPE_TYPES.BLOG]: {
      w: number
      h: number
      nodeId: string
      title: string
      url: string
      date: string
      description?: string
    }
    [SHAPE_TYPES.PROJECT]: {
      w: number
      h: number
      nodeId: string
      title: string
      url: string
      thumbnail?: string
      date: string
      tech?: string
    }
    [SHAPE_TYPES.MILESTONE]: {
      w: number
      h: number
      nodeId: string
      title: string
      date: string
      institution?: string
      description?: string
    }
  }
}

// Derived shape types for use in shape utils
export type HubShape = TLShape<typeof SHAPE_TYPES.HUB>
export type YouTubeNodeShape = TLShape<typeof SHAPE_TYPES.YOUTUBE>
export type BlogNodeShape = TLShape<typeof SHAPE_TYPES.BLOG>
export type ProjectNodeShape = TLShape<typeof SHAPE_TYPES.PROJECT>
export type MilestoneNodeShape = TLShape<typeof SHAPE_TYPES.MILESTONE>
```
