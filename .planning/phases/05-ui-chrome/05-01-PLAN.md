**Critical @theme token naming rules (from RESEARCH.md Pitfall 1):**
- Colors: Use `--color-*` prefix (generates text-*, bg-*, border-* utilities)
- Spacing: Use `--spacing-*` prefix (generates p-*, m-*, gap-* utilities)  
- Font sizes: Use `--font-size-*` prefix (generates text-* utilities)
- Font families: Use `--font-*` prefix (generates font-* utilities)
- Shadows: Use `--shadow-*` prefix (generates shadow-* utilities)
- Border radius: Use `--radius-*` prefix (generates rounded-* utilities)
- Z-index: Use `--z-index-*` prefix (generates z-* utilities)

**Token conversion examples:**
- `--text-primary: #FFFFFF` → `--color-primary: #FFFFFF` (use as `text-primary`, `bg-primary`)
- `--text-secondary: #A1A1AA` → `--color-secondary: #A1A1AA`
- `--interactive-default: #E0AFFF` → `--color-interactive: #E0AFFF`
- `--surface-default: #131313` → `--color-surface: #131313`
- `--glass-bg: rgba(28, 28, 28, 0.7)` → keep as CSS variable (not @theme, used in utility)

import { clsx, type ClassValue } from 'clsx'
import { twMerge } from 'tailwind-merge'

export function cn(...inputs: ClassValue[]) {
  return twMerge(clsx(inputs))
}

import * as React from 'react'
import * as DialogPrimitive from '@radix-ui/react-dialog'
import { cn } from '@/lib/utils'

const Dialog = DialogPrimitive.Root
const DialogTrigger = DialogPrimitive.Trigger
const DialogPortal = DialogPrimitive.Portal
const DialogClose = DialogPrimitive.Close

const DialogOverlay = React.forwardRef<
  React.ElementRef<typeof DialogPrimitive.Overlay>,
  React.ComponentPropsWithoutRef<typeof DialogPrimitive.Overlay>
>(({ className, ...props }, ref) => (
  <DialogPrimitive.Overlay
    ref={ref}
    className={cn(
      'fixed inset-0 z-[500] bg-black/60 backdrop-blur-sm',
      'data-[state=open]:animate-in data-[state=closed]:animate-out',
      'data-[state=closed]:fade-out-0 data-[state=open]:fade-in-0',
      className
    )}
    {...props}
  />
))
DialogOverlay.displayName = DialogPrimitive.Overlay.displayName

const DialogContent = React.forwardRef<
  React.ElementRef<typeof DialogPrimitive.Content>,
  React.ComponentPropsWithoutRef<typeof DialogPrimitive.Content>
>(({ className, children, ...props }, ref) => (
  <DialogPortal>
    <DialogOverlay />
    <DialogPrimitive.Content
      ref={ref}
      className={cn(
        'fixed left-1/2 top-1/2 z-[500] -translate-x-1/2 -translate-y-1/2',
        'w-[90vw] max-w-full md:max-w-lg md:w-auto',
        'glass rounded-2xl p-8',
        'data-[state=open]:animate-in data-[state=closed]:animate-out',
        'data-[state=closed]:fade-out-0 data-[state=open]:fade-in-0',
        'data-[state=closed]:zoom-out-95 data-[state=open]:zoom-in-95',
        'data-[state=closed]:slide-out-to-left-1/2 data-[state=closed]:slide-out-to-top-[48%]',
        'data-[state=open]:slide-in-from-left-1/2 data-[state=open]:slide-in-from-top-[48%]',
        'duration-200',
        className
      )}
      {...props}
    >
      {children}
    </DialogPrimitive.Content>
  </DialogPortal>
))
DialogContent.displayName = DialogPrimitive.Content.displayName

const DialogHeader = ({
